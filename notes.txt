TeamViewer dependencies check:
dpkg --info ./teamviewer.deb | grep Depends
OR
dpkg --info ./teamviewer.deb | \
    awk 'BEGIN{print "Depends:"} \
         /Depends: / { gsub("Depends: ", ""); \
                       n=split($0,deps,","); \
                       for(i=1;i<=n;i++) print deps[i] }'
